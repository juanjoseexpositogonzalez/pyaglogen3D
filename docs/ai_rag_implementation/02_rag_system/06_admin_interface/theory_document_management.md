```tsx
                <p className="text-sm text-muted-foreground">
                    Error: {document.metadata.error}
                </p>
                <div className="mt-4 flex gap-2">
                    <Button onClick={() => retryIngestion(document.id)}>
                        Retry
                    </Button>
                    <Button
                        variant="destructive"
                        onClick={() => deleteDocument(document.id)}
                    >
                        Delete
                    </Button>
                </div>
            </CardContent>
        </Card>
    );
}
```
